Algoritmo di Bellman Ford

1 Algoritmi preliminari

/*
G = grafo
s = nodo sorgente
*/
INIT_SS(G, s){
    for each u in V[G]{
        d[u] = +inf
        π[u] = NIL
    }
    d[s] = 0
}

/*
v = nodo di arrivo
u = nodo precedente a v (?)
w(u, v) = peso dell'arco tra u e v
*/
RELAX(u, v, w(u, v)){
    if(d[v] > d[u] + w(u, v)){
        d[v] = d[u] + w(u, v)
        π[v] = u
    }
}

2 Algoritmo

/*
G = grafo
w = funzione peso
s = nodo sorgente

PREREQUISITI: supporta pesi negativi ma non cicli negativi.
*/
BELLMAN-FORD(G, w, s){
    //INIZIALIZZAZIONE
    INIT_SS(G, s)

    //RELAX SU TUTTI I NODI (TRANNE LA SORGENTE)
    for i = 1 TO |V[G]| - 1{
        for each (u, v) in E[G]{
            RELAX(u, v, (u, v))
        }
    }

    //VERIFICA CHE NON CI SIANO CICLI NEGATIVI
    for each (u, v) in E[G]{
        if(d[v] > d[u] + w(u, v)){
            return (FALSE, d, π)
        }
    }

    return (TRUE, d, π)
}
